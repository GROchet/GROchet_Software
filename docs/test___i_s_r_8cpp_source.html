29void NotifyBtn() {
30 BaseType_t xHigherPriorityTaskWoken = pdFALSE;
31
32 bool up_state = digitalRead(BTN_PIN_UP) == LOW;
33 bool down_state = digitalRead(BTN_PIN_DOWN) == LOW;
34 bool left_state = digitalRead(BTN_PIN_LEFT) == LOW;
35 bool right_state = digitalRead(BTN_PIN_RIGHT) == LOW;
36 bool ok_state = digitalRead(BTN_PIN_OK) == LOW;
37
38 // OK overrides everything
39 if (ok_state) {
40 xEventGroupSetBitsFromISR(inputEventGroup, EVT_BTN_OK, &xHigherPriorityTaskWoken);
41 xEventGroupClearBitsFromISR(inputEventGroup, EVT_BTN_UP | EVT_BTN_DOWN | EVT_BTN_LEFT | EVT_BTN_RIGHT);
42 } else {
43 xEventGroupClearBitsFromISR(inputEventGroup, EVT_BTN_OK);
44
45 if (up_state) xEventGroupSetBitsFromISR(inputEventGroup, EVT_BTN_UP, &xHigherPriorityTaskWoken);
46 else xEventGroupClearBitsFromISR(inputEventGroup, EVT_BTN_UP);
47
48 if (down_state) xEventGroupSetBitsFromISR(inputEventGroup, EVT_BTN_DOWN, &xHigherPriorityTaskWoken);
49 else xEventGroupClearBitsFromISR(inputEventGroup, EVT_BTN_DOWN);
50
51 if (left_state) xEventGroupSetBitsFromISR(inputEventGroup, EVT_BTN_LEFT, &xHigherPriorityTaskWoken);
52 else xEventGroupClearBitsFromISR(inputEventGroup, EVT_BTN_LEFT);
53
54 if (right_state) xEventGroupSetBitsFromISR(inputEventGroup, EVT_BTN_RIGHT, &xHigherPriorityTaskWoken);
55 else xEventGroupClearBitsFromISR(inputEventGroup, EVT_BTN_RIGHT);
56 }
57
58 if (xHigherPriorityTaskWoken) {
59 portYIELD_FROM_ISR();
60 }
61}
