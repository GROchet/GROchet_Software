125GripperData gripperReceive(uint32_t timeoutMs = 500) {
126 GripperData data;
127 data.valid = false;
128
129 uint32_t start = millis();
130
131 while (SerialGripper.available() < 8) {
132 if (millis() - start > timeoutMs) return data;
133 vTaskDelay(pdMS_TO_TICKS(5));
134 }
135
136 uint8_t frame[8];
137 SerialGripper.readBytes(frame, 8);
138
139 // CRC check
140 if (frame[7] != crc8(frame, 7)) {
141 return data; // invalid
142 }
143
144 // Parse
145 data.response = frame[0];
146
147 data.openPos = frame[1] | (frame[2] << 8);
148 data.closedPos = frame[3] | (frame[4] << 8);
149 data.actualPos = frame[5] | (frame[6] << 8);
150
151 data.valid = true;
152 return data;
153}

314 if(bits & EVT_BTN_LEFT) deltaY += speed;

322 if(posX + deltaX > MAX_POS_X) deltaX = MAX_POS_X - posX;
323 if(posY + deltaY > MAX_POS_Y) deltaY = MAX_POS_Y - posY;
324 if(posX + deltaX < 0) deltaX = -posX;
325 if(posY + deltaY < 0) deltaY = -posY;
326
327 // Update positions
328 posX += deltaX;
329 posY += deltaY;
330
331 // CoreXY mapping
332 long targetA = posX + posY;
333 long targetB = posX - posY;
334
335 MOT_A.moveTo(targetA);
336 MOT_B.moveTo(targetB);
337 }
338
339 // always update steppers
340 MOT_A.run();
341 MOT_B.run();
342
343 // Check if we received a new notification telling us to STOP
344 if (ulTaskNotifyTake(pdTRUE, 0)) {
345 break; // Exit manual mode immediately
346 }
347 vTaskDelay(pdMS_TO_TICKS(5));

352void TaskStateControl (void *pvParameters) {
353 (void) pvParameters;
354
355 for(;;) {
356
357 switch(currentState) {
358 case DIFF_CHOOSE:
359 xTaskNotifyGive(motorTaskHandle); // Enable manual control
360 xEventGroupWaitBits(inputEventGroup, EVT_BTN_OK, pdTRUE, pdFALSE, portMAX_DELAY);
361 xTaskNotifyGive(motorTaskHandle); // Send stop signal
362 currentState = IDLE;
363 break;
364 case IDLE://DONE
365 Serial.println("IDLE: Use buttons to move, OK to calibrate.");
366 xTaskNotifyGive(motorTaskHandle); // Enable manual control
367 xEventGroupWaitBits(inputEventGroup, EVT_BTN_OK, pdTRUE, pdFALSE, portMAX_DELAY);
368 xTaskNotifyGive(motorTaskHandle); // Send stop signal
369 Serial.println("Transition to LOWERING");
370 currentState = LOWERING;
371 break;
372
373 case LOWERING:
374 //Séquence de mouvement vers le bas
375 MOT_Z.setSpeed(speed);
376 while(1) {
377 if (xEventGroupGetBits(inputEventGroup) & EVT_BTN_OK) break;
378 else if (MOT_Z.currentPosition() >= maxDownZPos) break; // On s'assure de pas descendre plus que la position levée, au cas où le limit switch ne marche pas
379 MOT_Z.runSpeed(); // actually step the motor
380 vTaskDelay(pdMS_TO_TICKS(2));
381 }
382 xEventGroupClearBits(inputEventGroup, EVT_BTN_OK); // clear the bit
383 MOT_Z.stop();
384
385 Serial.println("Transition to CLOSING");
386 currentState = CLOSING;
387 break;
388
389 case CLOSING:{
390 xSemaphoreTake(gripperMutex, portMAX_DELAY);
391 gripperSend(0x02);
392 //Attendre que le toutou soit attrapé ou pas (message de retour de OpenRB-150)
393 uint8_t resp1 = 0x04; //Start as "moving"
394 while((resp1 & 0x01) == 0 && (resp1 & 0x02) == 0){ //tant que ni toutou attrapé ni rien attrapé
395 gripperSend(0x04);
396 //Receive
397 GripperData g = gripperReceive();
398 if (!g.valid) {
399 // timeout or CRC fail
400 continue;
401 }
402 uint8_t resp1 = g.response;
403
404 // Wait for response with timeout
405 vTaskDelay(pdMS_TO_TICKS(50));
406 }
407 xSemaphoreGive(gripperMutex);
408
409 if (resp1 & 0x01) {
410 Serial.println("Toutou attrapé !");
411 xSemaphoreTake(toutouMutex, portMAX_DELAY);
412 toutouAttrape = true;
413 xSemaphoreGive(toutouMutex);
414 //On passe a lifting. MAIS ON DOIT CONTINUER DE DETECTER TOUTOU EN MEME TEMPS, JUSQUA DROPPING... créer une nouvelle tache ?
415 xTaskNotifyGive(detectToutouTaskHandle); // Lancer la tache de detection du toutou attrapé ou pas
416 }
417 else if (resp1 & 0x02) {
418 Serial.println("Rien attrapé.");
419 xSemaphoreTake(toutouMutex, portMAX_DELAY);
420 toutouAttrape = false;
421 xSemaphoreGive(toutouMutex);
422 //Variable pour PAS call ouvrirPince tantot.
423 }
424 currentState = LIFTING;
425 break;
426 }
427 case LIFTING:
428 //WAIT FOR Z TO BE LIFTED, THEN MOVE TO DROPZONE
429 MOT_Z.setSpeed(-speed);
430 while(MOT_Z.currentPosition() > liftedZPos){
431 MOT_Z.runSpeed(); // actually step the motor
432 vTaskDelay(pdMS_TO_TICKS(5));
433 }
434 MOT_Z.stop();
435 currentState = MOVING_TO_DROPZONE;
436 break;
437
438 case MOVING_TO_DROPZONE: //DONE
439 //Séquence de mouvement vers la dropzone
440 homeXY();
441 currentState = DROPPING;
442 break;
443
444 case DROPPING:{ //DONE;
445
446 xSemaphoreTake(toutouMutex, portMAX_DELAY);
447 bool local = toutouAttrape;
448 xSemaphoreGive(toutouMutex);
449
450 if (local) {
451 xSemaphoreTake(gripperMutex, portMAX_DELAY);
452 gripperSend(0x01); // ouvrir pince
453 xSemaphoreGive(gripperMutex);
454
455 vTaskDelay(pdMS_TO_TICKS(1000));
456 xTaskNotifyGive(detectToutouTaskHandle); // Stop la tache de detection du toutou, et ouvrir la pince si elle est encore fermée
457 }
458 xSemaphoreTake(toutouMutex, portMAX_DELAY);
459 toutouAttrape = false;
460 xSemaphoreGive(toutouMutex);
461 currentState = IDLE;
462 break;
463 }
464 }
465
466 vTaskDelay(pdMS_TO_TICKS(20));
467 }

579uint8_t crc8 (uint8_t *data, uint8_t len){
580 uint8_t crc = 0x00;
581 for (uint8_t i = 0;i<len;i++){
582 crc ^= data[i];
583 for (uint8_t b = 0;b<8;b++){
584 if(crc & 0x80) crc = (crc << 1) ^ 0x07;
585 else crc <<= 1;
586 }
587 }
588 return crc;
589}

596uint8_t gripperReceive(uint32_t timeoutMs = 500) {
597 uint32_t start = millis();
598 while (SerialGripper.available() < 8) { // Wait for 8 bytes (response + CRC + position data)
599 if (millis() - start > timeoutMs) return 0xFF; // timeout
600 vTaskDelay(pdMS_TO_TICKS(10));
601 }
602 uint8_t resp = SerialGripper.read();
603 uint8_t crc = SerialGripper.read();
604 if (crc != crc8(&resp, 1)) return 0xFF; // CRC mismatch
605 return resp;
606}
