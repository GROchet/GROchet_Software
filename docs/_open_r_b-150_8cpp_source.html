52void loop() {
53 // Wait until we have 2 bytes (msg + CRC)
54 if (Serial3.available() < 2) return;
55
56 uint8_t msg = Serial3.read();
57 uint8_t rxCrc = Serial3.read();
58
59 // Validate CRC — drop bad frames silently
60 if (rxCrc != crc8(&msg, 1)) return;
61
62 uint8_t response = 0;
63
64 // ── Act on bits ──
65 if (msg & 0x01) ouvrirPince();
66 if (msg & 0x02) fermerPince();
67 if (msg & 0x08) {
68 calibrer();
69 response |= 0x08; // Calibration done
70 }
71
72 if (msg & 0x04) { //4 = bit qui dit de verifier toutou
73 int status = detecterToutou();
74 if (status == 1) response |= 0x01; // toutou_attrape
75 if (status == 2) {
76 ouvrirPince(); // éviter de staller le moteur
77 response |= 0x02; // rien attrapé
78 }
79 if (status == 0) response |= 0x04; // en mouvement (ni toutou attrapé ni rien attrapé)
80 }
81
82 // ── Send response ──
83 ACTUAL_POS = dxl.getPresentPosition(id); // On lit la position actuelle à chaque requete pour l'envoyer dans la réponse, comme ça on a toujours la position à jour dans Unity sans avoir besoin d'une requete spécifique pour ça
84 uint8_t frame[8];
85 frame[0] = response;
86
87 // OPEN_POS
88 frame[1] = OPEN_POS & 0xFF;
89 frame[2] = (OPEN_POS >> 8) & 0xFF;
90
91 // CLOSED_POS
92 frame[3] = CLOSED_POS & 0xFF;
93 frame[4] = (CLOSED_POS >> 8) & 0xFF;
94
95 // ACTUAL_POS
96 frame[5] = ACTUAL_POS & 0xFF;
97 frame[6] = (ACTUAL_POS >> 8) & 0xFF;
98
99 // CRC
100 frame[7] = crc8(frame, 7);
101
102 Serial3.write(frame, 8);
103}

136int detecterToutou() {
137 static uint32_t stallSince = 0; // temps depuis la première détection de stall
138
139 int32_t pos = dxl.getPresentPosition(id);
140 int32_t vel = dxl.getPresentVelocity(id);
141 int16_t current = (int16_t)dxl.readControlTableItem(ControlTableItem::PRESENT_CURRENT, id);
142
143 if (pos >= CLOSED_POS - 5) {
144 stallSince = 0;
145 return 2; // fermé au complet, rien attrapé
146 }
147
148 if (abs(current) >= STALL_CURRENT && abs(vel) < VEL_THRESHOLD) {
149 if (stallSince == 0) stallSince = millis(); // premiere détection de stall
150
151 if (millis() - stallSince >= STALL_CONFIRM) {
152 stallSince = 0;
153 return 1; // stall confirmé = toutou attrapé
154 }
155 } else {
156 stallSince = 0; // condition broke, reset timer
157 }
158
159 return 0; // Encore en mouvement, pas encore de conclusion sur toutou attrapé ou pas
160}

162uint8_t crc8 (uint8_t *data, uint8_t len){
163 uint8_t crc = 0x00;
164 for (uint8_t i = 0;i<len;i++){
165 crc ^= data[i];
166 for (uint8_t b = 0;b<8;b++){
167 if(crc & 0x80) crc = (crc << 1) ^ 0x07;
168 else crc <<= 1;
169 }
170 }
171 return crc;
172}
